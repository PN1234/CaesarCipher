    System.out.print("Encryption for plaintext is:");
do {
for(j=0;j<Plaintext.length();j++)
    {
       for(i=0;i<str.length();i++)
    	{
    	if(Plaintext.charAt(j)==str.charAt(i)) {
    	
    	 System.out.print(str.charAt((str.indexOf(str.charAt(i))+3)%26));	 
    	   count++;
    		break;
        
    	}
    	}
       }

break;
} 
while(count!=Plaintext.length()-1);
